第一种：思想（创建10个桶存放最大值的最大位上分别为0123456789的数，对每组数进行插入排序，然后按顺序赋回去）（自己写）（*推荐*）
C语言，升序
#include<stdlib.h>
void bucketSort(int array[],int size){
	int i,j,k,temp1,temp2,temp3,count=0;
	int* countarray[10];
	int top[10];
	int max=array[0];
	// 初始化桶
	for(i=0;i<10;i++){
		countarray[i]=(int*)malloc(sizeof(int)*(size));
		top[i]=0;
	}
	// 获取原序列最大值
	for(i=1;i<size;i++){
		max=array[i]>max?array[i]:max;
	}
	// 获取原序列最大值位数
	while(max>0){
		max/=10;
		count++;
	}
	// 遍历原序列
	for(i=0;i<size;i++){
		temp1=count-1;
		temp2=array[i];
		while(temp1>0){
			temp2/=10;
			temp1--;
		}
		temp3=temp2%10;// 拿到了对应位的数字
		countarray[temp3][top[temp3]++]=array[i];// 对应桶的尾指针处赋值
	}
	// 对每个桶进行插入排序
	for(i=0;i<10;i++){
		for(j=1;j<top[i];j++){
			temp2=countarray[i][j];
			for(k=j;k>0&&countarray[i][k-1]>temp2;k--){
				countarray[i][k]=countarray[i][k-1];
			}
			countarray[i][k]=temp2;
		}
	}
	// 赋值回原序列
	temp1=0;
	for(i=0;i<10;i++){
		for(temp2=0;temp2<top[i];temp2++){
			array[temp1++]=countarray[i][temp2];
		}
		top[i]=0;
	}
	// 释放桶
	for(i=0;i<10;i++){
		free(countarray[i]);
	}
}
